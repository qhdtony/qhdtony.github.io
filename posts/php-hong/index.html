    在C里面， 宏是 C 语言的 预处理器指令（#define）。 编译前，预处理器会把宏替换成对应的代码文本。 宏：SQUARE(x) <=转换成=> 实际： ((5)*(5))

    我理解，就是一类方法经常用，建立的接口标准，方便外部调用. 也可以理解为语法糖，中间层。

    #define PI 3.14159
#define SQUARE(x) ((x)*(x))
printf("%f\n", SQUARE(5)); // 实际展开成 ((5)*(5))

    在PHP里面内核中的宏作用 用来简化zval操作、隐藏结构细节。否则每次写zval都得操作内部结构，太复杂也容易出错。

    0、回顾一下zval结构(简化)。

    typedef struct _zval_struct {
    zend_value        value;      // 变量的值 (联合体)
    union {
        struct {
            ZEND_ENDIAN_LOHI_4(
                zend_uchar    type,   // 类型信息 IS_LONG/IS_STRING/...
                zend_uchar    type_flags,
                zend_uchar    const_flags,
                zend_uchar    reserved
            )
        } v;
        uint32_t type_info;
    } u1;
    union {
        uint32_t     next;        // GC链表/HashTable索引
        uint32_t     cache_slot;  // opcode缓存槽
        uint32_t     lineno;      // 常量声明行号
    } u2;
} zval;

    typedef union _zend_value {
    zend_long         lval;       // 整数
    double            dval;       // 浮点数
    zend_refcounted  *counted;    // 引用计数类型 (字符串、数组、对象等)
    zend_string      *str;        // 字符串
    zend_array       *arr;        // 数组
    zend_object      *obj;        // 对象
    zend_resource    *res;        // 资源
    ...
} zend_value;

    1、简化 zval操作

    #define Z_TYPE(zval)      ((zval).u1.v.type)
#define Z_LVAL(zval)      ((zval).value.lval)
#define Z_STRVAL(zval)    (Z_STR(zval)->val)
#define Z_STRLEN(zval)    (Z_STR(zval)->len)Z_TYPE(z) 其实就是取 z.u1.v.type 字段
Z_LVAL(z) 就是取 z.value.lval
Z_STRVAL(z) 就是取字符串指针 z.value.str->val

    这样写可读性高。Z_TYPE(z) 对比 z.u1.v.type

    2、隐藏结构细节 Z_LVAL(z) 最终会转成 zval.value.lval 获取 long, 而不用管 zval.value.lval 简化代码，避免出错 提高兼容性，扩展者不用关心 PHP5/7 内部差异，宏保证了统一接口 （PHP内核不同版本的 zval 结构可能改动，但是宏接口对外没变）

    PHP扩展 <===> 宏 <===> zval结构

    常见类型宏

    设置值

    ZVAL_LONG(zval *z, zend_long l);       // 设置为 long
ZVAL_DOUBLE(zval *z, double d);        // 设置为 double
ZVAL_STRING(zval *z, const char *s);   // 设置为字符串 (复制)
ZVAL_STR(zval *z, zend_string *s);     // 设置为已有 zend_string
ZVAL_TRUE(zval *z);
ZVAL_FALSE(zval *z);
ZVAL_NULL(zval *z);

    zval zv;
ZVAL_LONG(&zv, 42);

    获取值

    Z_TYPE(zval)        // 获取类型 (IS_LONG, IS_STRING, ...)
Z_LVAL(zval)        // 获取 long 值
Z_DVAL(zval)        // 获取 double 值
Z_STRVAL(zval)      // 获取字符串的 char* 指针
Z_STRLEN(zval)      // 获取字符串长度
Z_ARRVAL(zval)      // 获取 HashTable* (数组)
Z_OBJ(zval)         // 获取 zend_object*

    if (Z_TYPE(zv) == IS_LONG) {
  printf("值 = %ld\n", Z_LVAL(zv));
